string ( TIMESTAMP YmdHM_UTC_"%Y%m%d%H%M" UTC )

	)

	CPACK_PACKAGE_FILE_NAME
	"${PACKAGE_NAME}-${CPACK_PACKAGE_VERSION}-${YmdHM_UTC_}-git_${/TWX/CFG/GIT_HASH}"
